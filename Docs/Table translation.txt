Table: People
	Username varchar(32) unique PK
	Password varchar(64)
Table: Roles
	Username varchar(32) unique PK FK(Person)
	Admin Boolean default(FALSE), constraint(count(true) <= 1)
Table: User
	SSN int unique PK
	Username unique FK(roles, people)
	First_name varchar(32)
	Last_name varchar(64)
	Address varchar(256)
	Phone int
Table: Account
	Acct_Num int unique PK
	SSN int unique FK(User)
	Approved Boolean default(FALSE)
	Balance decimal(18, 2)
